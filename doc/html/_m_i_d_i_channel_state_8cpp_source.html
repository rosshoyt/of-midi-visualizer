102 void MIDIChannelState::tryAddNoteOff(int note) {
103  // We may want to continue to hear the note if the sustain pedal is down, so track it
104  if (sustained) {
105  // Move note to sustainedNotes
106 
107  // First, we should check if the note still exists in the map.
108  // If not, return. This prevents out of range memory access occuring
109  // when accessing deleted notes in the notesHeldDown map.
110  if (notesHeldDown.count(note) == 0) return;
111 
112  // Next get the note's earliest occuring NoteOn Velocity
113  int noteOnVel = notesHeldDown.at(note); // 'concurrent access is safe' @s`ee http://www.cplusplus.com/reference/map/map/at/
